 switch (other.gameObject.tag)
        {
            case "Food":
                Food food;
                if (food = other.GetComponent<Food>())
                {
                    if (food.MyColorType == StageColorType.Right)
                    {
                        TakeFood();
                    }
                    else
                    {
                        if (_hasFever) TakeFood();
                        else TakeDamage();
                    }
                    EatObject(food);
                }
                break;

            case "Trap":
                TakeDamage();
                TakeEvents.OnTrapTaked.Invoke();
                break;

            case "Crystal":
                TakeEvents.OnCrystalTaked.Invoke();
                break;
        }
